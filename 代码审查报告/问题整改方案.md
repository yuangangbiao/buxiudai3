```python
# -*- coding: utf-8 -*-
"""
数据库连接工具模块
"""
import os
import pymysql
import logging

logger = logging.getLogger(__name__)

def get_mysql_password() -> str:
    """获取MySQL密码"""
    password = os.getenv('MYSQL_PASSWORD', '')
    if not password:
        raise ValueError("MYSQL_PASSWORD 环境变量未设置")
    return password

def create_db_connection(host='localhost', database='steel_belt', **kwargs):
    """创建数据库连接"""
    return pymysql.connect(
        host=host,
        user='root',
        password=get_mysql_password(),
        database=database,
        **kwargs
    )
```
