**修复方案**:

```javascript
// 在文件顶部添加 escHtml 函数
function escHtml(str) {
  if (str === null || str === undefined) return '';
  if (typeof str !== 'string') str = String(str);
  var div = document.createElement('div');
  div.appendChild(document.createTextNode(str));
  return div.innerHTML;
}

// L1046-1048 修复后

  '<div class="item-detail">'+escHtml(d.product_name)+'</div>'+
```

### 问题 3: eval 执行用户输入（trigger_config.html）

**级别**: 🟡 MEDIUM

**位置**: `trigger_config.html:469-488`

**当前代码**:
```javascript
function evaluateCondition(condition, context) {
    try {
        let expr = condition;

        for (const [key, value] of Object.entries(context)) {
            if (typeof value === 'string') {
                expr = expr.replace(new RegExp('\\b' + key + '\\b', 'g'), "'" + value + "'");
            } else {
                expr = expr.replace(new RegExp('\\b' + key + '\\b', 'g'), String(value));
            }
        }

        expr = expr.replace(/ AND /g, ' && ').replace(/ OR /g, ' || ');
        expr = expr.replace(/ == /g, ' === ').replace(/ != /g, ' !== ');

        return eval(expr);  // ⚠️ 危险
    } catch (e) {
        console.error('条件评估错误:', e);
        return false;
    }
}
```

**修复方案**:

```javascript
function evaluateCondition(condition, context) {
    try {
        let expr = condition;

        // 安全检查：只允许字母数字下划线和逻辑运算符
        if (!/^[\w\s=<>!&|().]+$/.test(expr)) {
            throw new Error('非法表达式字符');
        }

        // 变量替换
        for (const [key, value] of Object.entries(context)) {
            if (typeof value === 'string') {
                expr = expr.replace(new RegExp('\\b' + key + '\\b', 'g'), "'" + value.replace(/'/g, "\\'") + "'");
            } else {
                expr = expr.replace(new RegExp('\\b' + key + '\\b', 'g'), String(value));
            }
        }

        // 替换逻辑运算符
        expr = expr.replace(/\bAND\b/g, '&&').replace(/\bOR\b/g, '||');
        expr = expr.replace(/==/g, '===').replace(/!=/g, '!==');

        // 使用 Function 构造器（比 eval 稍安全，但仍有风险）
        // 更好的方案：使用表达式解析库
        const fn = new Function('return !!(' + expr + ')');
        return fn();
    } catch (e) {
        console.error('条件评估错误:', e);
        return false;
    }
}
```

**工作量**: 30 分钟

### 问题 5: 违规调用不阻止执行（wechat_work_bot_bp.py）

**级别**: 🟡 MEDIUM

**位置**: `wechat_work_bot_bp.py:495-501`

**当前代码**:
```python
def send_text(self, text):
    _violation_forbidden_bot('UserBot.AppBotProxy.send_text', ...)
    from wechat_app_bot import get_app_bot
    app_bot = get_app_bot()
    if app_bot:
        app_bot.send_text_to_user(self.target_user, text)  # ⚠️ 仍执行
```

**问题**: _violation_forbidden_bot 只写日志，不阻止执行

**修复方案**:

```python
def send_text(self, text):
    _violation_forbidden_bot('UserBot.AppBotProxy.send_text',
        'AppBotProxy.send_text 直接调用，应改用 wechat_msg_dispatcher.send_templated()')
    # ✅ 添加阻止逻辑
    return  # 直接返回，不执行后续代码
    # 或者抛出异常
    # raise RuntimeError('AppBotProxy.send_text 被禁用，请使用 wechat_msg_dispatcher.send_templated()')
```
