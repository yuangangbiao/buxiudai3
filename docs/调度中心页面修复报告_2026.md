```javascript
// 在文件顶部添加 escHtml 函数（如果没有）
function escHtml(str) {
  if (typeof str !== 'string') return String(str || '');
  var div = document.createElement('div');
  div.appendChild(document.createTextNode(str));
  return div.innerHTML;
}

// L1046-1048 修复后
```

### 问题 4: eval 条件评估风险（trigger_config.html）

**位置**: `trigger_config.html:469-488`

**当前代码**:
```javascript
function evaluateCondition(condition, context) {
  try {
    let expr = condition;
    expr = expr.replace(/ AND /g, ' && ').replace(/ OR /g, ' || ');
    expr = expr.replace(/ == /g, ' === ').replace(/ != /g, ' !== ');
    return eval(expr);  // ⚠️ 潜在风险
  } catch (e) {
    console.error('条件评估错误:', e);
    return false;
  }
}
```

**修复方案**:
```javascript
function evaluateCondition(condition, context) {
  try {
    let expr = condition;

    // 验证表达式只包含允许的字符
    if (!/^[\w\s=<>!&|().]+$/.test(expr)) {
      throw new Error('非法表达式字符');
    }

    // 替换逻辑运算符
    expr = expr.replace(/\bAND\b/g, '&&').replace(/\bOR\b/g, '||');
    expr = expr.replace(/==/g, '===').replace(/!=/g, '!==');

    // 使用 Function 构造器代替 eval（更安全）
    const fn = new Function('context', `
      with (context) {
        return !!((${expr}));
      }
    `);
    return fn(context);
  } catch (e) {
    console.error('条件评估错误:', e);
    return false;
  }
}
```
