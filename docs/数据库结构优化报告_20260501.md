## 高优先级问题

### 1. 缺乏独立的客户管理表

**问题严重性：** 🔴 高  
**影响范围：** 订单管理、客户分析、销售报表  
**当前实现：** 客户信息直接存储在 orders.customer_name 字段中

**问题描述：**
- 无法管理客户历史订单
- 客户信息重复存储，数据冗余
- 无法跟踪客户信用和付款情况
- 缺乏客户分类和分组管理

**优化建议：**

```sql
-- 新建客户表
CREATE TABLE customers (
    id INT PRIMARY KEY AUTO_INCREMENT,
    customer_code VARCHAR(20) UNIQUE NOT NULL COMMENT '客户编码',
    name VARCHAR(100) NOT NULL COMMENT '客户名称',
    contact_person VARCHAR(50) COMMENT '联系人',
    phone VARCHAR(20) COMMENT '联系电话',
    mobile VARCHAR(20) COMMENT '手机',
    address VARCHAR(255) COMMENT '地址',
    customer_group VARCHAR(50) COMMENT '客户分组',
    credit_limit DECIMAL(12,2) DEFAULT 0 COMMENT '信用额度',
    payment_days INT DEFAULT 30 COMMENT '账期天数',
    tax_rate DECIMAL(5,2) DEFAULT 0 COMMENT '税率',
    bank_name VARCHAR(100) COMMENT '开户银行',
    bank_account VARCHAR(50) COMMENT '银行账号',
    salesperson VARCHAR(50) COMMENT '负责业务员',
    status VARCHAR(20) DEFAULT '正常' COMMENT '状态：正常/停用',
    remark TEXT COMMENT '备注',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX idx_customer_code (customer_code),
    INDEX idx_customer_name (name),
    INDEX idx_salesperson (salesperson),
    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='客户信息表';

-- 订单表关联客户
ALTER TABLE orders 
ADD COLUMN customer_id INT COMMENT '客户ID',
ADD COLUMN salesperson VARCHAR(50) COMMENT '业务员',
ADD COLUMN contact_person VARCHAR(50) COMMENT '联系人',
ADD COLUMN priority_level VARCHAR(10) DEFAULT '中' COMMENT '优先级：高/中/低',
ADD COLUMN cancel_reason TEXT COMMENT '取消原因',
ADD COLUMN order_source VARCHAR(20) DEFAULT '线下' COMMENT '订单来源';

-- 添加外键索引
ALTER TABLE orders 
ADD INDEX idx_customer_id (customer_id),
ADD INDEX idx_salesperson (salesperson),
ADD INDEX idx_priority (priority_level);
```

### 3. 工序记录不完整

**问题严重性：** 🔴 高  
**影响范围：** 生产效率统计、计件工资、生产追溯

- 缺乏暂停/恢复记录
- 缺乏报废和返工数据
- 缺乏工序效率计算
- 缺乏机台关联

**优化建议：**

```sql
-- process_records 表新增字段
ALTER TABLE process_records 
ADD COLUMN planned_start DATETIME COMMENT '计划开始时间',

ADD COLUMN actual_pause_minutes INT DEFAULT 0 COMMENT '暂停总时长(分钟)',

    id INT PRIMARY KEY AUTO_INCREMENT,

    pause_start DATETIME NOT NULL COMMENT '暂停开始时间',
```

### 4. 库存表结构设计问题

**问题严重性：** 🔴 高  
**影响范围：** 库存管理、物料追溯、MRP计算

**当前问题：**
- inventory 表是独立设计，与主数据库不统一
- 缺乏库存预留和可用库存概念
- 缺乏库位管理
- 缺乏批次追踪

**优化建议：**

```sql
-- 新建原材料库存表（统一设计）
CREATE TABLE raw_material_inventory (
    id INT PRIMARY KEY AUTO_INCREMENT,
    material_code VARCHAR(50) UNIQUE NOT NULL COMMENT '物料编码',
    material_name VARCHAR(100) NOT NULL COMMENT '物料名称',
    material_type VARCHAR(50) COMMENT '物料类型',
    specification VARCHAR(100) COMMENT '规格',
    unit VARCHAR(20) DEFAULT 'kg' COMMENT '单位',
    current_stock DECIMAL(12,2) DEFAULT 0 COMMENT '当前库存',
    reserved_stock DECIMAL(12,2) DEFAULT 0 COMMENT '预留库存',
    available_stock DECIMAL(12,2) DEFAULT 0 COMMENT '可用库存',
    warehouse_id INT COMMENT '仓库ID',
    location VARCHAR(50) COMMENT '库位',
    min_stock DECIMAL(12,2) DEFAULT 0 COMMENT '最小库存(预警线)',
    max_stock DECIMAL(12,2) DEFAULT 0 COMMENT '最大库存(上限)',
    unit_price DECIMAL(12,2) DEFAULT 0 COMMENT '单价',
    last_inbound_date DATE COMMENT '最后入库日期',
    last_outbound_date DATE COMMENT '最后出库日期',
    status VARCHAR(20) DEFAULT '正常' COMMENT '状态',
    remark TEXT COMMENT '备注',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_material_code (material_code),
    INDEX idx_material_name (material_name),
    INDEX idx_warehouse (warehouse_id),
    INDEX idx_status (status),
    FOREIGN KEY (warehouse_id) REFERENCES warehouses(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='原材料库存表';

-- 新建库存流水表（增强）
CREATE TABLE inventory_transactions (
    id INT PRIMARY KEY AUTO_INCREMENT,
    trans_no VARCHAR(50) UNIQUE NOT NULL COMMENT '流水号',
    trans_type ENUM('inbound','outbound','transfer','adjust','return') NOT NULL COMMENT '流水类型',
    material_id INT NOT NULL COMMENT '物料ID',
    warehouse_id INT NOT NULL COMMENT '仓库ID',
    qty DECIMAL(12,2) NOT NULL COMMENT '数量',
    before_stock DECIMAL(12,2) COMMENT '变更前库存',
    after_stock DECIMAL(12,2) COMMENT '变更后库存',
    order_id INT COMMENT '关联订单ID',
    production_id INT COMMENT '关联生产工单ID',
    supplier_id INT COMMENT '供应商ID(入库)',
    batch_no VARCHAR(50) COMMENT '批次号',
    expire_date DATE COMMENT '有效期',
    operator VARCHAR(50) COMMENT '操作人',
    trans_date DATE NOT NULL COMMENT '业务日期',
    remark TEXT COMMENT '备注',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_trans_date (trans_date),
    INDEX idx_material_id (material_id),
    INDEX idx_order_id (order_id),
    INDEX idx_batch_no (batch_no),
    FOREIGN KEY (material_id) REFERENCES raw_material_inventory(id),
    FOREIGN KEY (warehouse_id) REFERENCES warehouses(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='库存流水表';

-- 批次库存表（用于批次追踪）
CREATE TABLE batch_inventory (
    id INT PRIMARY KEY AUTO_INCREMENT,
    material_id INT NOT NULL COMMENT '物料ID',
    batch_no VARCHAR(50) NOT NULL COMMENT '批次号',
    warehouse_id INT NOT NULL COMMENT '仓库ID',
    location VARCHAR(50) COMMENT '库位',
    qty DECIMAL(12,2) NOT NULL COMMENT '批次数量',
    unit_price DECIMAL(12,2) COMMENT '单价',
    received_date DATE COMMENT '收货日期',
    expire_date DATE COMMENT '有效期',
    supplier_id INT COMMENT '供应商ID',
    status VARCHAR(20) DEFAULT '正常' COMMENT '状态',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY uk_material_batch (material_id, batch_no),
    INDEX idx_material_id (material_id),
    INDEX idx_batch_no (batch_no),
    INDEX idx_warehouse (warehouse_id),
    FOREIGN KEY (material_id) REFERENCES raw_material_inventory(id),
    FOREIGN KEY (warehouse_id) REFERENCES warehouses(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='批次库存表';
```

```sql

ALTER TABLE orders 
ADD COLUMN is_deleted TINYINT(1) DEFAULT 0,
ADD COLUMN deleted_at DATETIME,

-- production_orders 表
ALTER TABLE production_orders 
ADD COLUMN is_deleted TINYINT(1) DEFAULT 0,
ADD COLUMN deleted_at DATETIME,
ADD COLUMN deleted_by VARCHAR(50);

-- 所有查询必须添加 WHERE is_deleted = 0
-- 或创建视图代替直接查询
CREATE VIEW v_orders AS

CREATE VIEW v_production_orders AS 
SELECT * FROM production_orders WHERE is_deleted = 0;
```

```python
# 文件名：migration_p0_phase1.py
# 阶段一：P0 优化迁移脚本

"""
阶段一优化：
1. 新增客户管理表
2. orders 表新增关键字段
3. 添加关键索引
"""

def migrate_p0_phase1():

        ("updated_by", "VARCHAR(50)", None, None),

            field_type=field_type,
            default_value=default_val,
            after_field=after_field

    ]
    
    for table, idx_name, columns in add_indexes:
        try:
            with migrator._get_cursor() as cursor:
                col_list = ", ".join([f"`{c}`" for c in columns])
                cursor.execute(f"ALTER TABLE `{table}` ADD INDEX `{idx_name}` ({col_list})")

    migrator.close()
    print("✅ 阶段一迁移完成！")

if __name__ == "__main__":
```
