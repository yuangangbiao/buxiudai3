## 一、方案概述

### 1.1 项目背景

**现有系统：**
- 不锈钢网带跟单系统3.1（主软件）：负责订单管理、生产跟踪
- 微信报工系统（mobile_api_ai）：负责微信端消息处理、报工指令

**当前问题：**
- 数据不同步：主软件与微信系统数据独立
- 流程不闭环：报工后任务单数据未减少
- 日志分散：主软件、云服务器日志分离
- 安全性不足：API访问无认证机制

### 1.2 方案目标

基于**本地-云端分离架构**，实现微信报工系统与不锈钢网带跟单系统3.1的深度融合，在保证功能完整性、安全性和高可用性的同时，最大限度降低云服务器依赖和运营成本。

### 1.3 核心原则

| 原则 | 说明 |
|------|------|
| **数据真实** | 时钟同步、哈希校验、无数据漂移 |
| **容错受限** | 重试限制、熔断机制、失败率<1% |
| **审计完整** | 所有操作必须记录、记录不可篡改 |
| **本地优先** | 核心业务全部部署在本地 |
| **云端轻量** | 仅承担消息转发和微信交互 |

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""云端消息转发服务（含增强API签名校验、Nonce校验、重放防护）"""

import os
import time
import hashlib
import logging
from flask import Flask, request, jsonify, abort
import requests
from requests.adapters import HTTPAdapter
from urllib3.util.retry import Retry

app = Flask(__name__)

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('/opt/cloud_forwarder/logs/forwarder.log'),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)

LOCAL_SERVER = os.getenv('LOCAL_SERVER', 'http://本地IP:5003')
API_SECRET_KEY = os.getenv('API_SECRET_KEY', 'change_me_in_production')

session = requests.Session()
retry_strategy = Retry(total=3, backoff_factor=1, status_forcelist=[500, 502, 503, 504])
adapter = HTTPAdapter(max_retries=retry_strategy)
session.mount("http://", adapter)


class EnhancedAPISignature:
    """增强版API签名校验器（含Nonce校验、请求体哈希）"""

    def __init__(self, redis_client=None, secret_key=None, timestamp_tolerance=300):
        self.redis_client = redis_client
        self.secret_key = secret_key or os.getenv('API_SECRET_KEY')
        self.timestamp_tolerance = timestamp_tolerance
        self.nonce_ttl = 600  # Nonce有效期10分钟

    def validate_signature(self, req):
        """增强版签名校验流程"""
        timestamp = req.headers.get('X-Timestamp')
        nonce = req.headers.get('X-Nonce')
        signature = req.headers.get('X-Signature')
        body = req.get_data()

        # 1. 参数完整性校验
        if not all([timestamp, nonce, signature]):
            logger.warning("API签名参数缺失")
            return False, "缺少签名参数"

        # 2. 时间戳校验（5分钟内有效）
        try:
            timestamp_int = int(timestamp)
        except ValueError:
            return False, "时间戳格式错误"

        if abs(time.time() - timestamp_int) > self.timestamp_tolerance:
            logger.warning(f"API签名时间戳过期")
            return False, "时间戳已过期"

        # 3. Nonce重复使用校验（防止重放攻击）
        if self.redis_client:
            nonce_key = f"api_nonce:{nonce}"
            if self.redis_client.exists(nonce_key):
                logger.warning(f"Nonce已被使用: {nonce}")
                return False, "Nonce无效或已使用"
            self.redis_client.setex(nonce_key, self.nonce_ttl, "1")

        # 4. 请求体哈希校验
        body_hash = hashlib.sha256(body).hexdigest()

        # 5. 签名计算（包含请求体哈希）
        sign_str = f"{timestamp}{nonce}{body_hash}{self.secret_key}"
        expected_signature = hashlib.sha256(sign_str.encode()).hexdigest()

        if signature != expected_signature:
            logger.warning(f"API签名校验失败")
            return False, "签名校验失败"

        return True, None


@app.route('/api/wechat/hook', methods=['POST'])
def wechat_hook():
    """接收微信回调并转发"""
    logger.info(f"收到微信回调: {request.args.get('msg_signature', '')[:8]}...")

    try:
        resp = session.post(
            f"{LOCAL_SERVER}/api/wechat/hook",
            params=request.args,
            data=request.get_data(),
            headers={'Content-Type': 'application/xml'},
            timeout=30
        )
        return resp.content, resp.status_code
    except requests.exceptions.Timeout:
        logger.error("转发超时")
        return jsonify({"code": 504, "message": "转发超时"}), 504
    except requests.exceptions.ConnectionError:
        logger.error("本地服务不可达")
        return jsonify({"code": 503, "message": "本地服务不可达"}), 503
    except Exception as e:
        logger.error(f"转发失败: {e}")
        return jsonify({"code": 500, "message": str(e)}), 500


@app.route('/api/sync/<path:subpath>', methods=['POST', 'GET'])
def forward_sync_api(subpath):
    """转发同步API（含增强签名校验）"""
    if request.method == 'POST':
        # 初始化签名校验器（需要Redis连接）
        redis_client = None
        try:
            import redis
            redis_client = redis.Redis(host='localhost', port=6379, decode_responses=True)
        except:
            pass

        validator = EnhancedAPISignature(redis_client=redis_client)
        is_valid, error_msg = validator.validate_signature(request)

        if not is_valid:
            logger.warning(f"签名校验失败: {error_msg}")
            return jsonify({"code": 401, "message": f"API签名校验失败: {error_msg}"}), 401

    try:
        url = f"{LOCAL_SERVER}/api/sync/{subpath}"
        resp = session.request(
            method=request.method,
            url=url,
            params=request.args,
            json=request.get_json(silent=True),
            headers={k: v for k, v in request.headers if k.lower() != 'host'},
            timeout=30
        )
        return resp.content, resp.status_code
    except Exception as e:
        logger.error(f"API转发失败: {e}")
        return jsonify({"code": 500, "message": str(e)}), 500


@app.route('/health', methods=['GET'])
def health_check():
    """健康检查"""
    cloud_status = "ok"
    local_status = "unknown"

    try:
        resp = session.get(f"{LOCAL_SERVER}/api/sync/health", timeout=5)
        local_status = "ok" if resp.status_code == 200 else "error"
    except:
        local_status = "unreachable"

    status = "ok" if (cloud_status == "ok" and local_status == "ok") else "degraded"
    return jsonify({
        "status": status,
        "cloud": cloud_status,
        "local": local_status
    }), 200 if status == "ok" else 503


if __name__ == '__main__':

### 5.1 时钟同步服务

**文件路径：** `/opt/wechat_bot/clock_sync.py`

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""时钟同步服务 - 防止数据漂移"""

import os
import time
import logging
import ntplib
from threading import Lock

logger = logging.getLogger(__name__)


class ClockSync:
    """时钟同步服务"""

    def __init__(self, ntp_server='ntp.aliyun.com'):
        self.ntp_server = ntp_server
        self.last_sync_time = 0
        self.offset = 0
        self.lock = Lock()

    def sync_time(self):
        """同步系统时间，偏差超过5秒则校正"""
        try:
            client = ntplib.NTPClient()
            response = client.request(self.ntp_server, version=3)
            self.offset = response.offset

            with self.lock:
                self.last_sync_time = time.time()

            if abs(self.offset) > 5:
                self._adjust_system_time(response.tx_time)
                logger.info(f"时间同步完成，偏移量: {self.offset:.2f}秒")
            else:
                logger.debug(f"时间正常，偏移量: {self.offset:.2f}秒")

            return True
        except Exception as e:
            logger.error(f"时间同步失败: {e}")
            return False

    def get_synced_time(self):
        """获取同步后的时间戳"""
        with self.lock:
            return time.time() + self.offset

    def _adjust_system_time(self, ntp_time):
        """调整系统时间"""
        if os.name == 'nt':
            import win32api
            win32api.SetSystemTime(*time.localtime(ntp_time))
        else:
            os.system(f"date -s @{int(ntp_time)}")


# 全局实例
clock_sync = ClockSync()
```

# -*- coding: utf-8 -*-
"""容错边界控制模块"""

import time
import logging

logger = logging.getLogger(__name__)


class FaultTolerance:
    """容错边界控制"""

    def __init__(self):
        self.config = {
            'max_retries': 3,
            'retry_delay': 1,
            'max_failure_rate': 0.01,
            'circuit_breaker_threshold': 50,
            'circuit_breaker_timeout': 30,
            'max_queue_depth': 1000
        }
        self.failure_count = 0
        self.total_count = 0
        self.circuit_state = 'closed'
        self.last_failure_time = 0

    def should_retry(self, attempt):
        """判断是否应该重试"""
        if attempt >= self.config['max_retries']:
            return False
        return True

    def record_success(self):
        """记录成功"""
        self.total_count += 1
        if self.circuit_state == 'half-open':
            self.circuit_state = 'closed'
        self.failure_count = max(0, self.failure_count - 1)

    def record_failure(self):
        """记录失败"""
        self.total_count += 1
        self.failure_count += 1
        self.last_failure_time = time.time()

        failure_rate = self.failure_count / max(1, self.total_count)

        if failure_rate > self.config['max_failure_rate']:
            self.circuit_state = 'open'
            logger.warning(
                f"失败率{failure_rate:.2%}超过阈值{self.config['max_failure_rate']:.2%}，熔断开启"
            )

    def is_circuit_open(self):
        """判断熔断是否开启"""
        if self.circuit_state == 'open':
            if time.time() - self.last_failure_time > self.config['circuit_breaker_timeout']:
                self.circuit_state = 'half-open'
                return False
            return True
        return False

    def get_failure_rate(self):
        """获取当前失败率"""
        return self.failure_count / max(1, self.total_count)

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""增强版审计日志系统 - 含IP记录、请求追踪、数据快照、不可篡改链"""

import os
import json
import time
import uuid
import random
import hashlib
import logging
from datetime import datetime
from elasticsearch import Elasticsearch
from threading import Lock
from contextlib import contextmanager

logger = logging.getLogger(__name__)


class EnhancedAuditLogger:
    """增强版审计日志系统（强制记录+不可篡改+全链路追踪）"""

    def __init__(self, es_hosts=None, redis_client=None):
        if es_hosts is None:
            es_hosts = ['localhost:9200']
        self.es = Elasticsearch(es_hosts)
        self.redis_client = redis_client
        self.fallback_path = '/opt/audit_log/fallback'
        self.chain = []
        self.chain_lock = Lock()
        self.request_context = {}
        os.makedirs(self.fallback_path, exist_ok=True)

    def _ensure_index(self):
        """确保索引存在（含完整mapping）"""
        index_name = f"audit-logs-{datetime.now().strftime('%Y.%m')}"
        if not self.es.indices.exists(index=index_name):
            mapping = {
                "mappings": {
                    "properties": {
                        "timestamp": {"type": "date"},
                        "operation_type": {"type": "keyword"},
                        "user_id": {"type": "keyword"},
                        "user_name": {"type": "keyword"},
                        "action": {"type": "keyword"},
                        "result": {"type": "keyword"},
                        "details": {"type": "object"},
                        "client_ip": {"type": "ip"},           # 新增：客户端IP
                        "request_id": {"type": "keyword"},     # 新增：请求追踪ID
                        "trace_id": {"type": "keyword"},       # 新增：链路追踪ID
                        "user_agent": {"type": "text"},        # 新增：用户代理
                        "data_before": {"type": "object"},     # 新增：操作前快照
                        "data_after": {"type": "object"},      # 新增：操作后快照
                        "audit_id": {"type": "keyword"},
                        "block_hash": {"type": "keyword"},
                        "previous_hash": {"type": "keyword"},
                        "chain_index": {"type": "long"},
                        "signature": {"type": "keyword"}        # 新增：签名
                    }
                }
            }
            self.es.indices.create(index=index_name, body=mapping)
            logger.info(f"创建审计索引: {index_name}")

    def _generate_audit_id(self):
        """生成唯一审计ID"""
        return f"AUD{datetime.now().strftime('%Y%m%d%H%M%S')}{random.randint(1000, 9999)}"

    def _generate_request_id(self):
        """生成请求追踪ID"""
        return f"REQ{uuid.uuid4().hex[:16].upper()}"

    def _calculate_block_hash(self, block):
        """计算区块哈希"""
        data = {
            'timestamp': block.get('timestamp', ''),
            'audit_id': block.get('audit_id', ''),
            'operation_type': block.get('operation_type', ''),
            'user_id': block.get('user_id', ''),
            'action': block.get('action', ''),
            'result': block.get('result', ''),
            'client_ip': block.get('client_ip', ''),
            'request_id': block.get('request_id', ''),
            'previous_hash': block.get('previous_hash', '')
        }
        raw = json.dumps(data, sort_keys=True, ensure_ascii=False)
        return hashlib.sha256(raw.encode()).hexdigest()

    def log(self, operation_type, user_id, user_name, action, result,
            details=None, client_ip=None, request_id=None, trace_id=None,
            user_agent=None, data_before=None, data_after=None):
        """记录审计日志（强制写入）"""
        with self.chain_lock:
            block = {
                'timestamp': datetime.now().isoformat(),
                'operation_type': operation_type,
                'user_id': user_id,
                'user_name': user_name,
                'action': action,
                'result': result,
                'details': details or {},
                'client_ip': client_ip or '0.0.0.0',
                'request_id': request_id or self._generate_request_id(),
                'trace_id': trace_id or '',
                'user_agent': user_agent or '',
                'data_before': data_before,
                'data_after': data_after,
                'audit_id': self._generate_audit_id(),
                'previous_hash': self.chain[0]['block_hash'] if self.chain else '0' * 64,
                'chain_index': len(self.chain)
            }

            block['block_hash'] = self._calculate_block_hash(block)

            try:
                self._write_to_es(block)
                self.chain.insert(0, block)
                logger.info(f"审计记录写入成功: {block['audit_id']}")
            except Exception as e:
                logger.error(f"ES写入失败，降级到本地文件: {e}")
                self._write_to_file(block)

    def _write_to_es(self, block):
        """写入Elasticsearch"""
        index_name = f"audit-logs-{datetime.now().strftime('%Y.%m')}"
        self.es.index(index=index_name, document=block)

    def _write_to_file(self, block):
        """降级写入本地文件"""
        file_path = f"{self.fallback_path}/audit_{datetime.now().strftime('%Y%m%d')}.log"
        with open(file_path, 'a', encoding='utf-8') as f:
            f.write(json.dumps(block, ensure_ascii=False) + '\n')

    def verify_chain(self):
        """验证链完整性"""
        with self.chain_lock:
            for i in range(1, len(self.chain)):
                current = self.chain[i]
                previous = self.chain[i - 1]

                if current['previous_hash'] != previous['block_hash']:
                    logger.error(f"审计链断裂 at index {i}")
                    return False, {'type': 'chain_break', 'index': i}

                expected_hash = self._calculate_block_hash(current)
                if current['block_hash'] != expected_hash:
                    logger.error(f"审计记录篡改 at index {i}")
                    return False, {'type': 'tamper_detected', 'index': i}

            logger.info("审计链完整性验证通过")
            return True, None


# 全局实例
audit_logger = None


def init_audit_logger(es_hosts=None, redis_client=None):
    """初始化全局审计日志器"""
    global audit_logger
    audit_logger = EnhancedAuditLogger(es_hosts=es_hosts, redis_client=redis_client)

        **kwargs

### 8.3 增强备份脚本

**文件路径：** `/opt/backup/enhanced_backup.py`

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""增强版备份脚本 - 含完整性校验、恢复验证、自动清理"""

import os
import sys
import json
import time
import subprocess
import shutil
import hashlib
import logging
from datetime import datetime
from pathlib import Path

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('/opt/backup/backup.log'),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)


class EnhancedBackupManager:
    """增强版备份管理器"""

    def __init__(self, backup_dir='/opt/backup', redis_password=None):
        self.backup_dir = Path(backup_dir)
        self.redis_password = redis_password or os.getenv('REDIS_PASSWORD')
        self.retention_days = 7
        self.backup_dir.mkdir(parents=True, exist_ok=True)
        self._ensure_directories()

    def _ensure_directories(self):
        """确保备份目录结构存在"""
        for subdir in ['redis', 'elasticsearch', 'configs', 'logs', 'metadata']:
            (self.backup_dir / subdir).mkdir(parents=True, exist_ok=True)

    def _calculate_file_hash(self, file_path):
        """计算文件SHA256哈希"""
        sha256_hash = hashlib.sha256()
        try:
            with open(file_path, 'rb') as f:
                for chunk in iter(lambda: f.read(8192), b''):
                    sha256_hash.update(chunk)
            return sha256_hash.hexdigest()
        except Exception as e:
            logger.error(f"计算文件哈希失败 {file_path}: {e}")
            return None

    def _check_disk_space(self):
        """检查磁盘空间是否充足"""
        try:
            result = shutil.disk_usage(self.backup_dir)
            available = result.free
            min_space = 10 * 1024 * 1024 * 1024  # 10GB

            if available < min_space:
                return False, {'available_gb': round(available / 1024**3, 2), 'status': 'insufficient'}
            return True, {'available_gb': round(available / 1024**3, 2), 'status': 'ok'}
        except Exception as e:
            return False, {'error': str(e)}

    def backup_redis(self):
        """备份Redis（BGSAVE + 校验）"""
        logger.info("开始Redis备份...")
        backup_info = {'type': 'redis', 'status': 'pending'}

        try:
            redis_dump_path = '/var/lib/redis/dump.rdb'
            if not os.path.exists(redis_dump_path):
                backup_info['status'] = 'skipped'
                return backup_info

            # BGSAVE
            bg_cmd = ['redis-cli'] + (['-a', self.redis_password] if self.redis_password else []) + ['BGSAVE']
            subprocess.run(bg_cmd, capture_output=True, timeout=30)
            time.sleep(2)

            # 复制备份
            backup_filename = f"redis_{datetime.now().strftime('%Y%m%d_%H%M%S')}.rdb"
            backup_path = self.backup_dir / 'redis' / backup_filename
            shutil.copy2(redis_dump_path, backup_path)

            # 计算哈希
            file_hash = self._calculate_file_hash(backup_path)

            # 保存哈希文件
            hash_file = backup_path.with_suffix('.rdb.sha256')
            with open(hash_file, 'w') as f:
                f.write(file_hash)

            backup_info.update({
                'status': 'success',
                'file': str(backup_path),
                'hash': file_hash,
                'size_bytes': os.path.getsize(backup_path)
            })
            logger.info(f"Redis备份完成: {backup_path}")

        except Exception as e:
            logger.error(f"Redis备份失败: {e}")
            backup_info.update({'status': 'failed', 'error': str(e)})

        return backup_info

    def verify_backup(self, backup_info):
        """验证备份完整性"""
        if backup_info.get('type') != 'redis':
            return True, {}

        backup_file = backup_info.get('file')
        expected_hash = backup_info.get('hash')

        if not backup_file or not os.path.exists(backup_file):
            return False, {'error': '备份文件不存在'}

        actual_hash = self._calculate_file_hash(backup_file)
        if actual_hash != expected_hash:
            return False, {'error': '哈希校验失败'}

        return True, {'verified': True, 'size': os.path.getsize(backup_file)}

    def test_restore(self, backup_type='redis'):
        """测试恢复功能"""
        if backup_type == 'redis':
            redis_backups = sorted((self.backup_dir / 'redis').glob('*.rdb'),
                                   key=lambda x: x.stat().st_mtime, reverse=True)
            if not redis_backups:
                return False, {'error': '无备份文件'}

            try:
                test_rdb = '/tmp/test_restore.rdb'
                shutil.copy2(redis_backups[0], test_rdb)
                os.remove(test_rdb)
                return True, {'file': str(redis_backups[0])}
            except Exception as e:
                return False, {'error': str(e)}

        return False, {'error': 'unsupported type'}

    def cleanup_old_backups(self):
        """清理过期备份"""
        cutoff = time.time() - (self.retention_days * 86400)
        cleaned = 0

        for subdir in ['redis', 'elasticsearch', 'configs']:
            for item in (self.backup_dir / subdir).iterdir():
                if item.is_file() and item.stat().st_mtime < cutoff:
                    item.unlink()
                    cleaned += 1
                elif item.is_dir() and item.stat().st_mtime < cutoff:
                    shutil.rmtree(item)
                    cleaned += 1

        logger.info(f"清理完成: 删除{cleaned}个文件")
        return {'cleaned': cleaned}

    def run_full_backup(self):
        """执行完整备份流程"""
        logger.info("=== 开始完整备份 ===")

        result = {
            'started_at': datetime.now().isoformat(),
            'disk_space': {},
            'redis': {},
            'verification': {},
            'cleanup': {},
            'overall_status': 'pending'
        }

        # 检查磁盘空间
        is_enough, space_info = self._check_disk_space()
        result['disk_space'] = space_info
        if not is_enough:
            result['overall_status'] = 'failed'
            return result

        # Redis备份
        redis_result = self.backup_redis()
        result['redis'] = redis_result

        # 验证备份
        if redis_result.get('status') == 'success':
            is_valid, verify_info = self.verify_backup(redis_result)
            result['verification']['redis'] = {'passed': is_valid, 'details': verify_info}

            if is_valid:
                can_restore, restore_info = self.test_restore('redis')
                result['verification']['restore_test'] = {'passed': can_restore, 'details': restore_info}

        # 清理
        cleanup_result = self.cleanup_old_backups()
        result['cleanup'] = cleanup_result

        result['completed_at'] = datetime.now().isoformat()
        result['overall_status'] = 'success' if redis_result.get('status') == 'success' else 'partial'

        # 保存元数据
        metadata_file = self.backup_dir / 'metadata' / f"backup_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
        with open(metadata_file, 'w') as f:
            json.dump(result, f, indent=2)

        logger.info(f"=== 备份完成: {result['overall_status']} ===")
        return result


if __name__ == '__main__':
    manager = EnhancedBackupManager()
    if len(sys.argv) > 1 and sys.argv[1] == 'backup':
        result = manager.run_full_backup()
        print(json.dumps(result, indent=2))
        sys.exit(0 if result['overall_status'] == 'success' else 1)
```
