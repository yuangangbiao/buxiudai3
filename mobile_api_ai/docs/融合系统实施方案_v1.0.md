## 一、方案概述

### 1.1 项目背景

**现有系统：**
- 不锈钢网带跟单系统3.1（主软件）：负责订单管理、生产跟踪
- 微信报工系统（mobile_api_ai）：负责微信端消息处理、报工指令

**当前问题：**
- 数据不同步：主软件与微信系统数据独立
- 流程不闭环：报工后任务单数据未减少
- 日志分散：主软件、云服务器日志分离
- 安全性不足：API访问无认证机制

### 1.2 方案目标

基于**本地-云端分离架构**，实现微信报工系统与不锈钢网带跟单系统3.1的深度融合，在保证功能完整性、安全性和高可用性的同时，最大限度降低云服务器依赖和运营成本。

### 1.3 核心原则

| 原则 | 说明 |
|------|------|
| **数据真实** | 时钟同步、哈希校验、无数据漂移 |
| **容错受限** | 重试限制、熔断机制、失败率<1% |
| **审计完整** | 所有操作必须记录、记录不可篡改 |
| **本地优先** | 核心业务全部部署在本地 |
| **云端轻量** | 仅承担消息转发和微信交互 |

### 3.3 云端转发服务

**文件路径：** `/opt/cloud_forwarder/cloud_forwarder.py`

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""云端消息转发服务（含API签名校验）"""

import os
import time
import hashlib
import logging
from flask import Flask, request, jsonify, abort
import requests
from requests.adapters import HTTPAdapter
from urllib3.util.retry import Retry

app = Flask(__name__)

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('/opt/cloud_forwarder/logs/forwarder.log'),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)

LOCAL_SERVER = os.getenv('LOCAL_SERVER', 'http://本地IP:5003')
API_SECRET_KEY = os.getenv('API_SECRET_KEY', 'change_me_in_production')

session = requests.Session()
retry_strategy = Retry(total=3, backoff_factor=1, status_forcelist=[500, 502, 503, 504])
adapter = HTTPAdapter(max_retries=retry_strategy)
session.mount("http://", adapter)


def validate_api_signature():
    """API签名校验：SHA256(timestamp + nonce + secret_key)"""
    timestamp = request.headers.get('X-Timestamp')
    nonce = request.headers.get('X-Nonce')
    signature = request.headers.get('X-Signature')
    
    if not all([timestamp, nonce, signature]):
        logger.warning("API签名参数缺失")
        return False
    
    if abs(time.time() - int(timestamp)) > 300:
        logger.warning("API签名时间戳过期")
        return False
    
    sign_str = f"{timestamp}{nonce}{API_SECRET_KEY}"
    expected_signature = hashlib.sha256(sign_str.encode()).hexdigest()
    
    if signature != expected_signature:
        logger.warning("API签名校验失败")
        return False
    
    return True


@app.route('/api/wechat/hook', methods=['POST'])
def wechat_hook():
    """接收微信回调并转发"""
    logger.info(f"收到微信回调: {request.args.get('msg_signature', '')[:8]}...")
    
    try:
        resp = session.post(
            f"{LOCAL_SERVER}/api/wechat/hook",
            params=request.args,
            data=request.get_data(),
            headers={'Content-Type': 'application/xml'},
            timeout=30
        )
        return resp.content, resp.status_code
    except requests.exceptions.Timeout:
        logger.error("转发超时")
        return jsonify({"code": 504, "message": "转发超时"}), 504
    except requests.exceptions.ConnectionError:
        logger.error("本地服务不可达")
        return jsonify({"code": 503, "message": "本地服务不可达"}), 503
    except Exception as e:
        logger.error(f"转发失败: {e}")
        return jsonify({"code": 500, "message": str(e)}), 500


@app.route('/api/sync/<path:subpath>', methods=['POST', 'GET'])
def forward_sync_api(subpath):
    """转发同步API（含签名校验）"""
    if request.method == 'POST' and not validate_api_signature():
        abort(401, description="API签名校验失败")
    
    try:
        url = f"{LOCAL_SERVER}/api/sync/{subpath}"
        resp = session.request(
            method=request.method,
            url=url,
            params=request.args,
            json=request.get_json(silent=True),
            headers={k: v for k, v in request.headers if k.lower() != 'host'},
            timeout=30
        )
        return resp.content, resp.status_code
    except Exception as e:
        logger.error(f"API转发失败: {e}")
        return jsonify({"code": 500, "message": str(e)}), 500


@app.route('/health', methods=['GET'])
def health_check():
    """健康检查"""
    cloud_status = "ok"
    local_status = "unknown"
    
    try:
        resp = session.get(f"{LOCAL_SERVER}/api/sync/health", timeout=5)
        local_status = "ok" if resp.status_code == 200 else "error"
    except:
        local_status = "unreachable"
    
    status = "ok" if (cloud_status == "ok" and local_status == "ok") else "degraded"
    return jsonify({
        "status": status,
        "cloud": cloud_status,
        "local": local_status
    }), 200 if status == "ok" else 503


if __name__ == '__main__':
    os.makedirs('/opt/cloud_forwarder/logs', exist_ok=True)
    app.run(host='0.0.0.0', port=5003, debug=False, threaded=True)
```

### 5.1 时钟同步服务

**文件路径：** `/opt/wechat_bot/clock_sync.py`

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""时钟同步服务 - 防止数据漂移"""

import os
import time
import logging
import ntplib
from threading import Lock

logger = logging.getLogger(__name__)


class ClockSync:
    """时钟同步服务"""
    
    def __init__(self, ntp_server='ntp.aliyun.com'):
        self.ntp_server = ntp_server
        self.last_sync_time = 0
        self.offset = 0
        self.lock = Lock()
    
    def sync_time(self):
        """同步系统时间，偏差超过5秒则校正"""
        try:
            client = ntplib.NTPClient()
            response = client.request(self.ntp_server, version=3)
            self.offset = response.offset
            
            with self.lock:
                self.last_sync_time = time.time()
            
            if abs(self.offset) > 5:
                self._adjust_system_time(response.tx_time)
                logger.info(f"时间同步完成，偏移量: {self.offset:.2f}秒")
            else:
                logger.debug(f"时间正常，偏移量: {self.offset:.2f}秒")
                
            return True
        except Exception as e:
            logger.error(f"时间同步失败: {e}")
            return False
    
    def get_synced_time(self):
        """获取同步后的时间戳"""
        with self.lock:
            return time.time() + self.offset
    
    def _adjust_system_time(self, ntp_time):
        """调整系统时间"""
        if os.name == 'nt':
            import win32api
            win32api.SetSystemTime(*time.localtime(ntp_time))
        else:
            os.system(f"date -s @{int(ntp_time)}")


# 全局实例
clock_sync = ClockSync()
```

### 5.2 数据完整性校验

**文件路径：** `/opt/wechat_bot/data_integrity.py`

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""数据完整性校验 - 防止数据漂移"""

import json
import hashlib


class DataIntegrity:
    """数据完整性校验"""
    
    @staticmethod
    def calculate_hash(data):
        """计算数据哈希值"""
        if isinstance(data, dict):
            sorted_data = json.dumps(data, sort_keys=True, ensure_ascii=False)
        else:
            sorted_data = str(data)
        return hashlib.sha256(sorted_data.encode()).hexdigest()
    
    @staticmethod
    def verify_hash(data, expected_hash):
        """验证数据完整性"""
        return DataIntegrity.calculate_hash(data) == expected_hash
    
    @staticmethod
    def create_data_fingerprint(order_no, process, quantity, timestamp):
        """创建数据指纹"""
        raw = f"{order_no}:{process}:{quantity}:{timestamp}"
        return hashlib.sha256(raw.encode()).hexdigest()
```

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""容错边界控制模块"""

import time
import logging

logger = logging.getLogger(__name__)


class FaultTolerance:
    """容错边界控制"""
    
    def __init__(self):
        self.config = {
            'max_retries': 3,                    # 最大重试次数
            'retry_delay': 1,                    # 重试间隔(秒)
            'max_failure_rate': 0.01,           # 最大失败率(1%)
            'circuit_breaker_threshold': 50,    # 熔断阈值
            'circuit_breaker_timeout': 30,      # 熔断超时(秒)
            'max_queue_depth': 1000              # 最大队列深度
        }
        self.failure_count = 0
        self.total_count = 0
        self.circuit_state = 'closed'
        self.last_failure_time = 0
    
    def should_retry(self, attempt):
        """判断是否应该重试"""
        if attempt >= self.config['max_retries']:
            return False
        return True
    
    def record_success(self):
        """记录成功"""
        self.total_count += 1
        if self.circuit_state == 'half-open':
            self.circuit_state = 'closed'
        self.failure_count = max(0, self.failure_count - 1)
    
    def record_failure(self):
        """记录失败"""
        self.total_count += 1
        self.failure_count += 1
        self.last_failure_time = time.time()
        
        failure_rate = self.failure_count / max(1, self.total_count)
        
        if failure_rate > self.config['max_failure_rate']:
            self.circuit_state = 'open'
            logger.warning(
                f"失败率{failure_rate:.2%}超过阈值{self.config['max_failure_rate']:.2%}，熔断开启"
            )
    
    def is_circuit_open(self):
        """判断熔断是否开启"""
        if self.circuit_state == 'open':
            if time.time() - self.last_failure_time > self.config['circuit_breaker_timeout']:
                self.circuit_state = 'half-open'
                return False
            return True
        return False
    
    def get_failure_rate(self):
        """获取当前失败率"""
        return self.failure_count / max(1, self.total_count)


# 全局实例
fault_tolerance = FaultTolerance()
```

## 七、审计日志系统

### 7.1 区块链式审计日志

**文件路径：** `/opt/audit_log/audit_logger.py`

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""审计日志系统 - 区块链式不可篡改审计"""

import os
import json
import time
import random
import hashlib
import logging
from datetime import datetime
from elasticsearch import Elasticsearch
from threading import Lock

logger = logging.getLogger(__name__)


class AuditLogger:
    """审计日志系统（强制记录+不可篡改）"""
    
    def __init__(self, es_hosts=None):
        if es_hosts is None:
            es_hosts = ['localhost:9200']
        self.es = Elasticsearch(es_hosts)
        self.fallback_path = '/opt/audit_log/fallback'
        self.chain = []
        self.chain_lock = Lock()
        os.makedirs(self.fallback_path, exist_ok=True)
        
        self._ensure_index()
        self._load_chain()
    
    def _ensure_index(self):
        """确保索引存在"""
        index_name = f"audit-logs-{datetime.now().strftime('%Y.%m')}"
        if not self.es.indices.exists(index=index_name):
            mapping = {
                "mappings": {
                    "properties": {
                        "timestamp": {"type": "date"},
                        "operation_type": {"type": "keyword"},
                        "user_id": {"type": "keyword"},
                        "user_name": {"type": "keyword"},
                        "action": {"type": "keyword"},
                        "result": {"type": "keyword"},
                        "details": {"type": "object"},
                        "audit_id": {"type": "keyword"},
                        "block_hash": {"type": "keyword"},
                        "previous_hash": {"type": "keyword"},
                        "chain_index": {"type": "long"}
                    }
                }
            }
            self.es.indices.create(index=index_name, body=mapping)
            logger.info(f"创建审计索引: {index_name}")
    
    def _load_chain(self):
        """加载审计链"""
        try:
            result = self.es.search(
                index="audit-logs-*",
                query={"match_all": {}},
                sort=[{"chain_index": {"order": "desc"}}],
                size=1000
            )
            
            if result['hits']['hits']:
                self.chain = [hit['_source'] for hit in result['hits']['hits']]
                logger.info(f"加载审计链，当前长度: {len(self.chain)}")
        except Exception as e:
            logger.warning(f"加载审计链失败: {e}")
            self.chain = []
    
    def _calculate_block_hash(self, block):
        """计算区块哈希"""
        data = {
            'timestamp': block.get('timestamp', ''),
            'audit_id': block.get('audit_id', ''),
            'operation_type': block.get('operation_type', ''),
            'action': block.get('action', ''),
            'previous_hash': block.get('previous_hash', '')
        }
        raw = json.dumps(data, sort_keys=True, ensure_ascii=False)
        return hashlib.sha256(raw.encode()).hexdigest()
    
    def _generate_audit_id(self):
        """生成唯一审计ID"""
        return f"AUD{datetime.now().strftime('%Y%m%d%H%M%S')}{random.randint(1000, 9999)}"
    
    def log(self, operation_type, user_id, user_name, action, result, details=None):
        """记录审计日志（强制写入）"""
        with self.chain_lock:
            block = {
                'timestamp': datetime.now().isoformat(),
                'operation_type': operation_type,
                'user_id': user_id,
                'user_name': user_name,
                'action': action,
                'result': result,
                'details': details or {},
                'audit_id': self._generate_audit_id(),
                'previous_hash': self.chain[0]['block_hash'] if self.chain else '0' * 64,
                'chain_index': len(self.chain)
            }
            
            block['block_hash'] = self._calculate_block_hash(block)
            
            try:
                self._write_to_es(block)
                self.chain.insert(0, block)
            except Exception as e:
                logger.error(f"ES写入失败，降级到本地文件: {e}")
                self._write_to_file(block)
    
    def _write_to_es(self, block):
        """写入Elasticsearch"""
        index_name = f"audit-logs-{datetime.now().strftime('%Y.%m')}"
        self.es.index(index=index_name, document=block)
    
    def _write_to_file(self, block):
        """降级写入本地文件"""
        file_path = f"{self.fallback_path}/audit_{datetime.now().strftime('%Y%m%d')}.log"
        with open(file_path, 'a', encoding='utf-8') as f:
            f.write(json.dumps(block, ensure_ascii=False) + '\n')
    
    def verify_chain(self):
        """验证链完整性"""
        with self.chain_lock:
            for i in range(1, len(self.chain)):
                current = self.chain[i]
                previous = self.chain[i - 1]
                
                if current['previous_hash'] != previous['block_hash']:
                    logger.error(f"审计链断裂 at index {i}")
                    return False
                
                expected_hash = self._calculate_block_hash(current)
                if current['block_hash'] != expected_hash:
                    logger.error(f"审计记录篡改 at index {i}")
                    return False
            
            logger.info("审计链完整性验证通过")
            return True
    
    def query_logs(self, user_id=None, operation_type=None, start_time=None, end_time=None):
        """查询审计日志"""
        must_clauses = []
        
        if user_id:
            must_clauses.append({"term": {"user_id": user_id}})
        if operation_type:
            must_clauses.append({"term": {"operation_type": operation_type}})
        if start_time or end_time:
            range_query = {}
            if start_time:
                range_query["gte"] = start_time
            if end_time:
                range_query["lte"] = end_time
            must_clauses.append({"range": {"timestamp": range_query}})
        
        query = {"bool": {"must": must_clauses}} if must_clauses else {"match_all": {}}
        
        try:
            result = self.es.search(
                index="audit-logs-*",
                query=query,
                sort=[{"timestamp": {"order": "desc"}}],
                size=1000
            )
            return [hit['_source'] for hit in result['hits']['hits']]
        except Exception as e:
            logger.error(f"查询审计日志失败: {e}")
            return []


# 全局实例
audit_logger = AuditLogger()


def log_operation(operation_type, user_id, user_name, action, result, details=None):
    """快捷记录函数"""
    audit_logger.log(operation_type, user_id, user_name, action, result, details)
```
