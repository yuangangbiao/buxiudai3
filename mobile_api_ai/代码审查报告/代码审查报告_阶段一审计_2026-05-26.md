```python
# utils/check_annotation_coverage.py（需创建）
"""检查项目中函数/方法的类型注解覆盖率"""
import ast
import os
import sys
from pathlib import Path

def check_annotation_coverage(filepath):
    with open(filepath, 'r', encoding='utf-8') as f:
        tree = ast.parse(f.read())
    functions = [n for n in ast.walk(tree) if isinstance(n, (ast.FunctionDef, ast.AsyncFunctionDef))]
    if not functions:
        return 100.0
    annotated = sum(1 for f in functions if f.returns or any(a.arg for a in f.args.args if a.annotation))
    return annotated / len(functions) * 100

if __name__ == '__main__':
    threshold = float(os.getenv('ANNOTATION_THRESHOLD', '50'))
    failed = []
    for pyfile in Path('.').rglob('*.py'):
        if 'tests' in pyfile.parts or '.venv' in pyfile.parts:
            continue
        rate = check_annotation_coverage(pyfile)
        if rate < threshold:
            failed.append((pyfile, rate))
    if failed:
        for f, r in failed:
            print(f'❌ {f}: {r:.1f}%')
        sys.exit(1)
    print(f'✅ 类型注解覆盖率检查通过（threshold={threshold}%）')
```
