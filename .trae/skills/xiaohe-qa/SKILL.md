**审查方法论**: "**假设代码全是错的,直到用源码证据证明它是对的**"

| 审查维度 | 审查项 | 证据要求 |

## 四、典型 P0 修复方案模板(可直接复用)

### 模板 1:状态机白名单校验

```python
# ⚠️ 重要:数据库中 status 字段存储的是中文字符串,必须使用中文字段!
# 查看数据库实际值: SELECT DISTINCT status FROM orders;

# 修复前(零校验,任意跳级)
def update_status(order_id, new_status):
    cur.execute("UPDATE orders SET status=%s WHERE id=%s", (new_status, order_id))

# 修复后(白名单 + 行锁)
ALLOWED_TRANSITIONS = {
    '已创建': ['已确认'],
    '已确认': ['已排产'],
    '已排产': ['生产中'],
    '生产中': ['已完工'],
    '已完工': ['已发货', '已归档'],
    '已发货': ['已归档'],
}

def update_status(order_id, new_status):
    with db.cursor() as cur:
        cur.execute("SELECT status FROM orders WHERE id=%s FOR UPDATE", (order_id,))
        row = cur.fetchone()
        if not row:
            return error("订单不存在")
        current = row['status']
        if new_status not in ALLOWED_TRANSITIONS.get(current, []):
            return error(f"不允许从 {current} 跳到 {new_status}")
        cur.execute("UPDATE orders SET status=%s WHERE id=%s", (new_status, order_id))

# 或者使用中英文字段映射(当代码中使用英文字段但数据库存中文时)
FIELD_MAP = {
    'created': '已创建',
    'confirmed': '已确认',
    'scheduled': '已排产',
    'in_progress': '生产中',
    'completed': '已完工',
    'shipped': '已发货',
    'archived': '已归档',
}
```

### 模板 2:边界校验(拒绝而非修正)

```python
# 修复前(用 max(0, int) 修正,丢失语义)
def update_defect_qty(qc_id, defect_qty):
    safe_qty = max(0, int(defect_qty))  # ❌ 负数被吞掉
    cur.execute("UPDATE quality_records SET defect_qty=%s WHERE id=%s", (safe_qty, qc_id))

# 修复后(拒绝 + 友好提示)
def update_defect_qty(qc_id, defect_qty):
    if defect_qty is None:
        return error("不良数不能为空")
    if not isinstance(defect_qty, int):
        return error("不良数必须为整数")
    if defect_qty < 0:
        return error("不良数不能为负数,请输入 0 或正整数")
    if defect_qty > 9999:
        return error("不良数超出限制(0-9999),请检查输入")
    cur.execute("UPDATE quality_records SET defect_qty=%s WHERE id=%s", (defect_qty, qc_id))
```

### 模板 3:运单号高并发去重(时间戳 → UUID)
